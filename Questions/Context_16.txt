// Beyonce portrayed which character in the film, "Cadillac Records"?
MATCH (b:Beyonce)-[:PORTRAYED]->(c:Character)-[:IN_FILM]->(f:Film {title: "Cadillac Records"})
RETURN c.name AS CharacterName

// Beyonce gave her entire salary from "Cadillac Records" to which organization?
MATCH (b:Beyonce)-[:DONATED_SALARY_TO]->(o:Organization)-[:FROM_FILM]->(f:Film {title: "Cadillac Records"})
RETURN o.name AS OrganizationName

// Which song did Beyonce sing at the first couple's inaugural ball?
MATCH (b:Beyonce)-[:SANG_AT]->(e:Event {name: "First Couple's Inaugural Ball"})-[:SONG_PERFORMED]->(s:Song)
RETURN s.title AS SongTitle

// What genre of film was the movie, "Obsessed," in which Beyonce starred?
MATCH (b:Beyonce)-[:STARRED_IN]->(f:Film {title: "Obsessed"})
RETURN f.genre AS FilmGenre

// A fight scene from the movie, "Obsessed," won which award for Beyonce?
MATCH (b:Beyonce)-[:WON]->(a:Award {category: "Fight Scene"})-[:FROM_FILM]->(f:Film {title: "Obsessed"})
RETURN a.name AS AwardName
