// "Charlie's Angels" featured which single from the band members?
MATCH (g:Group {name: "Destiny's Child"})-[:RELEASED]->(s:Single)-[:FEATURED_IN]->(f:Film {title: "Charlie's Angels"})
RETURN s.title AS FeaturedSingle

// How many weeks did their single "Independent Women Part I" stay on top?
MATCH (s:Single {title: "Independent Women Part I"})
RETURN s.top_duration_weeks AS WeeksOnTop

// For what network, did Beyonce land a major movie role in?
MATCH (b:Beyonce)-[:LANDED_MAJOR_ROLE_ON]->(n:Network)
RETURN n.name AS NetworkName

// Their third album, Survivor, sold how many during its first week?
MATCH (g:Group {name: "Destiny's Child"})-[:RELEASED]->(a:Album {title: "Survivor"})
RETURN a.first_week_sales AS FirstWeekSales

// What French composer wrote the original opera ''Carmen'' in the 19th century?
MATCH (o:Opera {title: "Carmen"})-[:WRITTEN_BY]->(c:Composer {nationality: "French", century: 19})
RETURN c.name AS ComposerName

// What album caused a lawsuit to be filed in 2001?
MATCH (a:Album)-[:CAUSED_LAWSUIT_IN]->(l:Lawsuit {year: 2001})
RETURN a.title AS AlbumTitle

// "Independent Women Part I" was on which 2000 film's soundtrack?
MATCH (s:Single {title: "Independent Women Part I"})-[:FEATURED_IN]->(f:Film)
WHERE f.release_year = 2000
RETURN f.title AS FilmTitle

// Which film did BeyoncÃ© star in 2001 with Mekhi Phifer?
MATCH (b:Beyonce)-[:STARRED_IN]->(f:Film)-[:CO_STARRED_WITH]->(a:Actor {name: "Mekhi Phifer"})
WHERE f.release_year = 2001
RETURN f.title AS FilmTitle

// What was the name of Destiny's Child's third album?
MATCH (g:Group {name: "Destiny's Child"})-[:RELEASED]->(a:Album)
WHERE a.is_third = true
RETURN a.title AS ThirdAlbumTitle

// Who filed a lawsuit over "Survivor"?
MATCH (a:Album {title: "Survivor"})-[:CAUSE_OF_LAWSUIT]->(l:Lawsuit)-[:FILED_BY]->(p:Person)
RETURN p.name AS LawsuitFiler

// When did Destiny's Child announce their hiatus?
MATCH (g:Group {name: "Destiny's Child"})
RETURN g.hiatus_announcement_year AS HiatusYear
