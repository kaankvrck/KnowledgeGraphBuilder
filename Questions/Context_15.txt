// Beyonce got married in 2008 to whom?
MATCH (b:Beyonce)-[:MARRIED_TO]->(p:Person)
WHERE b.marriage_year = 2008
RETURN p.name AS SpouseName

// Her third album, "I am...Sasha Fierce" was released when?
MATCH (a:Album {title: "I Am...Sasha Fierce"})
RETURN a.release_year AS ReleaseYear

// For which decade did Beyonce have more top ten songs than any other woman?
MATCH (b:Beyonce)-[:TOP_TEN_IN_DECADE]->(d:Decade)
RETURN d.name AS Decade

// Which singer beat out Beyonce for best video performance?
MATCH (a:Award {category: "Best Video Performance"})-[:BEAT]->(s:Singer)-[:BEAT]->(b:Beyonce)
RETURN s.name AS WinningSinger

// In 2009, Beyonce started her second world tour and grossed how much money?
MATCH (b:Beyonce)-[:TOURED]->(t:Tour {name: "Second World Tour", start_year: 2009})
RETURN t.revenue AS TourRevenue

// How did she reveal the marriage?
MATCH (b:Beyonce)-[:REVEALED_MARRIAGE_BY]->(m:Method)
RETURN m.description AS RevealMethod

// When did Beyonce begin her second world tour?
MATCH (b:Beyonce)-[:TOURED]->(t:Tour {name: "Second World Tour"})
RETURN t.start_year AS StartYear

// Who beat out Beyonce for Best Female Video?
MATCH (a:Award {category: "Best Female Video"})-[:BEAT]->(s:Singer)-[:BEAT]->(b:Beyonce)
RETURN s.name AS WinningSinger

// How much did the second world tour make in dollars?
MATCH (b:Beyonce)-[:TOURED]->(t:Tour {name: "Second World Tour"})
RETURN t.revenue AS TourRevenue

// When did Beyoncé get married?
MATCH (b:Beyonce)
RETURN b.marriage_year AS MarriageYear

// Who did Beyoncé marry?
MATCH (b:Beyonce)-[:MARRIED_TO]->(p:Person)
RETURN p.name AS SpouseName

// Who is Beyoncé's alter ego?
MATCH (b:Beyonce)-[:HAS_ALTER_EGO]->(ae:AlterEgo)
RETURN ae.name AS AlterEgoName

// The video for what song won Beyoncé the 2009 MTV Video of the Year award?
MATCH (s:Song)-[:WON]->(a:Award {name: "MTV Video of the Year", year: 2009})
RETURN s.title AS WinningSong

// Which prominent star felt the 2009 Female Video of the Year award should have gone to Beyoncé instead of Taylor Swift?
MATCH (p:ProminentStar)-[:DISAPPROVED]->(a:Award {category: "Female Video of the Year", year: 2009})
RETURN p.name AS ProminentStarName
