// Who did Beyonce star with in the movie, "Austin Powers in Goldmember"?
MATCH (b:Beyonce)-[:STARRED_WITH]->(a:Actor)-[:IN_MOVIE]->(m:Movie {title: "Austin Powers in Goldmember"})
RETURN a.name AS CoStar

// Which three countries did Beyonce's song "Work It Out" achieve top ten status?
MATCH (s:Song {title: "Work It Out"})-[:TOP_TEN_IN]->(c:Country)
RETURN c.name AS TopTenCountries

// Beyonce starred with Cuba Gooding Jr. in which film?
MATCH (b:Beyonce)-[:STARRED_WITH]->(a:Actor {name: "Cuba Gooding Jr."})-[:IN_MOVIE]->(f:Film)
RETURN f.title AS FilmTitle

// Who did Beyonce record the lead single with in the movie "The Fighting Temptations"?
MATCH (b:Beyonce)-[:RECORDED_LEAD_SINGLE_WITH]->(a:Artist)-[:FOR_MOVIE]->(m:Movie {title: "The Fighting Temptations"})
RETURN a.name AS CollaboratingArtist

// Which other song from the soundtrack did better in the charts?
MATCH (m:Movie {title: "The Fighting Temptations"})-[:HAS_SOUNDTRACK]->(s:Song)
WHERE s.chart_rank < 10
RETURN s.title AS BetterChartingSong
