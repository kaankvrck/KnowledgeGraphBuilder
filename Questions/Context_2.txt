// After her second solo album, what other entertainment venture did Beyonce explore?
MATCH (b:Beyonce)-[:RELEASED]->(a:Album)
WHERE a.is_second_solo = true
MATCH (b)-[:EXPLORED]->(e:EntertainmentVenture)
RETURN e.name AS EntertainmentVenture

// Which artist did Beyonce marry?
MATCH (b:Beyonce)-[:MARRIED_TO]->(a:Artist)
RETURN a.name AS ArtistName

// To set the record for Grammys, how many did Beyonce win?
MATCH (b:Beyonce)-[:WON]->(a:Award {name: "Grammy"})
RETURN count(a) AS GrammyCount

// For what movie did Beyonce receive her first Golden Globe nomination?
MATCH (b:Beyonce)-[:NOMINATED_FOR]->(a:Award {name: "Golden Globe", type: "first_nomination"})-[:FOR_MOVIE]->(m:Movie)
RETURN m.title AS MovieTitle

// When did Beyonce take a hiatus in her career and take control of her management?
MATCH (b:Beyonce)
RETURN b.hiatus_year AS HiatusYear, b.management_control_year AS ManagementControlYear

// Which album was darker in tone from her previous work?
MATCH (b:Beyonce)-[:RELEASED]->(a:Album)
WHERE a.is_darker_in_tone = true
RETURN a.title AS AlbumTitle

// After what movie portraying Etta James did Beyonce create Sasha Fierce?
MATCH (b:Beyonce)-[:PORTRAYED]->(r:Role {character: "Etta James"})-[:IN_MOVIE]->(m:Movie)
MATCH (b)-[:CREATED]->(ae:AlterEgo {name: "Sasha Fierce"})
RETURN m.title AS MovieTitle

// When did Destiny's Child end their group act?
MATCH (g:Group {name: "Destiny's Child"})
RETURN g.end_year AS EndYear

// What was the name of Beyoncé's second solo album?
MATCH (b:Beyonce)-[:RELEASED]->(a:Album)
WHERE a.is_second_solo = true
RETURN a.title AS AlbumTitle

// What was Beyoncé's first acting job, in 2006?
MATCH (b:Beyonce)-[:ACTED_IN]->(m:Movie {release_year: 2006})
RETURN m.title AS MovieTitle

// Who is Beyoncé married to?
MATCH (b:Beyonce)-[:MARRIED_TO]->(p:Person)
RETURN p.name AS PartnerName

// What is the name of Beyoncé's alter-ego?
MATCH (b:Beyonce)-[:CREATED]->(ae:AlterEgo)
RETURN ae.name AS AlterEgoName
