// In her music, what are some recurring elements in them?
MATCH (b:Beyonce)-[:RELEASED]->(a:Album)-[:CONTAINS]->(s:Song)-[:HAS_ELEMENT]->(e:Element)
RETURN DISTINCT e.name AS RecurringElements

// Time magazine named her one of the most 100 what people of the century?
MATCH (b:Beyonce)-[:NAMED_IN]->(l:List {magazine: "Time", name: "100 People of the Century"})
RETURN l.category AS Category

// Which magazine declared her the most dominant woman musician?
MATCH (b:Beyonce)-[:DECLARED_AS]->(t:Title {name: "Most Dominant Woman Musician"})-[:BY]->(m:Magazine)
RETURN m.name AS MagazineName

// In which decade did the Recording Industry Association of America recognize Beyonce as The Top Certified Artist?
MATCH (b:Beyonce)-[:RECOGNIZED_AS]->(t:Title {name: "Top Certified Artist"})-[:BY]->(a:Association {name: "Recording Industry Association of America"})
RETURN t.decade AS Decade

// What magazine rated Beyonce as the most powerful female musician in 2015?
MATCH (b:Beyonce)-[:DECLARED_AS]->(t:Title {name: "Most Powerful Female Musician"})-[:BY]->(m:Magazine)
WHERE t.year = 2015
RETURN m.name AS MagazineName

// How did Beyonce describe herself as a feminist?
MATCH (b:Beyonce)
RETURN b.feminist_description AS Description

// In which years did Time rate Beyonce in the 100 most influential people in the world?
MATCH (b:Beyonce)-[:NAMED_IN]->(l:List {magazine: "Time", name: "100 Most Influential People in the World"})
RETURN l.year AS Years

// How many records has Beyonce sold in her 19-year career?
MATCH (b:Beyonce)-[:SOLD]->(r:Record)
RETURN sum(r.quantity) AS TotalRecords

// How many records did Beyoncé sell as part of Destiny's Child?
MATCH (b:Beyonce)-[:MEMBER_OF]->(g:Group {name: "Destiny's Child"})-[:SOLD]->(r:Record)
RETURN sum(r.quantity) AS TotalRecordsWithDestinysChild

// After leaving Destiny's Child, how many records did Beyoncé release under her own name?
MATCH (b:Beyonce)-[:LEFT]->(g:Group {name: "Destiny's Child"})
MATCH (b)-[:RELEASED]->(r:Record)
RETURN count(r) AS SoloAlbums

// How many Grammy awards has Beyoncé won?
MATCH (b:Beyonce)-[:WON]->(a:Award {name: "Grammy"})
RETURN count(a) AS GrammyCount

// What magazine named Beyoncé as the most powerful female musician for 2015?
MATCH (b:Beyonce)-[:DECLARED_AS]->(t:Title {name: "Most Powerful Female Musician"})-[:BY]->(m:Magazine)
WHERE t.year = 2015
RETURN m.name AS MagazineName
