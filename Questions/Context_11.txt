// What was the highest Beyoncé's first solo recording achieved in the Billboard Hot 100?
MATCH (s:Song)-[:IN_ALBUM]->(a:Album {is_first_solo: true})-[:ACHIEVED_RANK_IN]->(c:Chart {name: "Billboard Hot 100"})
RETURN min(c.rank) AS HighestRank

// Beyoncé's first album by herself was called what?
MATCH (a:Album {is_first_solo: true})
RETURN a.title AS FirstAlbumTitle

// How many has "Dangerously in Love" sold worldwide since its debut?
MATCH (a:Album {title: "Dangerously in Love"})
RETURN a.worldwide_sales AS WorldwideSales

// Beyoncé's first number one song was which song?
MATCH (s:Song {is_first_number_one: true})-[:IN_ALBUM]->(a:Album {is_first_solo: true})
RETURN s.title AS FirstNumberOneSong

// How many top five singles came from her first album?
MATCH (s:Song)-[:IN_ALBUM]->(a:Album {is_first_solo: true})-[:ACHIEVED_RANK_IN]->(c:Chart {name: "Billboard Hot 100"})
WHERE c.rank <= 5
RETURN count(s) AS TopFiveSingles

// Beyoncé's first solo album in the U.S. with what artist in the lead single?
MATCH (s:Song)-[:IN_ALBUM]->(a:Album {is_first_solo: true})-[:FEATURES]->(ar:Artist)
RETURN ar.name AS LeadSingleArtist

// What solo album did Beyoncé release in 2003?
MATCH (a:Album {release_year: 2003, is_first_solo: true})
RETURN a.title AS SoloAlbum

// The album "Dangerously in Love" achieved what spot on the Billboard Top 100 chart?
MATCH (a:Album {title: "Dangerously in Love"})-[:ACHIEVED_RANK_IN]->(c:Chart {name: "Billboard Hot 100"})
RETURN c.rank AS ChartPosition

// "The Closer I Get to You" was recorded with which artist?
MATCH (s:Song {title: "The Closer I Get to You"})-[:FEATURES]->(ar:Artist)
RETURN ar.name AS CollaboratingArtist

// Which artist was associated with Beyoncé's premiere solo recording?
MATCH (b:Beyonce)-[:RELEASED_WITH]->(ar:Artist)
WHERE b.is_first_solo = true
RETURN ar.name AS AssociatedArtist

// When did Beyoncé release her first solo album?
MATCH (a:Album {is_first_solo: true})
RETURN a.release_year AS FirstSoloAlbumYear

// What is the lead single on Beyoncé's first album?
MATCH (s:Song {is_lead_single: true})-[:IN_ALBUM]->(a:Album {is_first_solo: true})
RETURN s.title AS LeadSingle

// Who helped Beyoncé earn a Grammy award for Best R&B Performance by a Duo or Group at the 46th annual Grammy Awards?
MATCH (b:Beyonce)-[:WON_WITH]->(ar:Artist)-[:FOR_AWARD]->(a:Award {name: "Grammy", category: "Best R&B Performance by a Duo or Group", year: 46})
RETURN ar.name AS CollaboratingArtist

// How many awards did Beyoncé win at the 46th Grammy Awards?
MATCH (b:Beyonce)-[:WON]->(a:Award {name: "Grammy", year: 46})
RETURN count(a) AS GrammyAwardsCount
