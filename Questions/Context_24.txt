// Where did Beyonce release her 5th album to a huge surprise?
MATCH (a:Album {is_fifth: true})-[:RELEASED_AT]->(p:Platform)
RETURN p.name AS ReleasePlatform

// When did Beyonce get her fifth consecutive number one hit album?
MATCH (a:Album {is_fifth: true})-[:ACHIEVED]->(c:Chart {name: "Number One Hit"})
RETURN c.date AS AchievementDate

// Where was the album released?
MATCH (a:Album {is_fifth: true})-[:RELEASED_AT]->(p:Platform)
RETURN p.name AS ReleasePlatform

// Who joined Beyonce on her "On The Run" tour?
MATCH (b:Beyonce)-[:TOURED_WITH]->(a:Artist)-[:IN_TOUR]->(t:Tour {name: "On The Run"})
RETURN a.name AS CollaboratingArtist

// Who reported Beyonce to be the top-earning woman in music?
MATCH (b:Beyonce)-[:REPORTED_BY]->(o:Organization)-[:FOR_EARNINGS]->(e:Earnings {category: "Top Earning Woman in Music"})
RETURN o.name AS ReportingOrganization

// How much more were her earnings than the year before?
MATCH (b:Beyonce)-[:EARNINGS]->(e1:Earnings {year: "Current Year"}), (e2:Earnings {year: "Previous Year"})
RETURN e1.amount - e2.amount AS IncreaseInEarnings

// When did Beyoncé release her fifth studio album?
MATCH (a:Album {is_fifth: true})
RETURN a.release_date AS ReleaseDate

// How many digital copies of her fifth album did Beyoncé sell in six days?
MATCH (a:Album {is_fifth: true})
RETURN a.six_day_digital_sales AS SixDayDigitalSales

// What song on Beyoncé's fifth studio album featured her husband?
MATCH (s:Song)-[:FEATURES]->(a:Artist {relationship: "husband"})-[:IN_ALBUM]->(al:Album {is_fifth: true})
RETURN s.title AS FeaturedSong

// What was the name of the tour featuring both Beyoncé and Jay Z?
MATCH (b:Beyonce)-[:TOURED_WITH]->(a:Artist {name: "Jay Z"})-[:IN_TOUR]->(t:Tour)
RETURN t.name AS TourName
