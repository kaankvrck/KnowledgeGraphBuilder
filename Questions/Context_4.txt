// Beyonce's younger sibling also sang with her in what band?
MATCH (b:Beyonce)-[:SIBLING_OF]->(s:Sibling)-[:SANG_WITH]->(g:Group)
RETURN g.name AS BandName

// Where did Beyonce get her name from?
MATCH (b:Beyonce)
RETURN b.origin_of_name AS Origin

// What race was Beyonce's father?
MATCH (f:Person {relationship: "father", child: "Beyonce"})
RETURN f.race AS Race

// Beyonce's childhood home believed in what religion?
MATCH (b:Beyonce)-[:GREW_UP_IN]->(h:Home)-[:BELIEVES_IN]->(r:Religion)
RETURN r.name AS Religion

MATCH (f:Person {relationship: "father", child: "Beyonce", job_title: "Sales Manager"})
